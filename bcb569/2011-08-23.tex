\section*{Introduction}

\begin{mitemize}
  \item structural informatics is about structures of life's basic elements: proteins, DNA, RNA
  \item by structure, we mean shapes, geometry, motion
\end{mitemize}

\subsection*{Scale concept}

\begin{mitemize}
  \item atom: about 1\AA
  \item amino acid: $< 1 nm$
  \item protein: $< 10 nm$
  \item cell: $10 \mu m$
\end{mitemize}
